class Foo {
    private double big = 3.2e+23; // Some sort of big value
    private double small = -4.70e-9; // Some sort of small value

    /**
     * Multiline comment
     * Starts the program
     */
    public void main(String[] args) {
        int size = 3;
        int [size] array = { 1, 2, 3 };
        int index = 0;
        while (index != 0) {
            index = index - 1;
            System.out.println(array[index] * big * small);
        }
        for (var num : array) {
            System.out.println(num * big * small);
        }
    }
}
